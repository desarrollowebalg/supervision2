# Para escuchar los comentarios guardados:

document.querySelector('comment-box').addEventListener('comment-saved', (event) => {
    const comment = event.detail;
    const comment = event.detail;
    console.log('ID del usuario:', comment.userId);
    console.log('Nombre del usuario:', comment.userName);
    console.log('Nickname:', comment.nickname);
    console.log('Texto del comentario:', comment.text);
    console.log('Fecha y hora:', comment.timestamp);  
});

# Si se requiere cambiar el texto del placeholder:

const commentBox = document.querySelector('comment-box');
commentBox.setPlaceholderText('Nuevo texto del placeholder');
